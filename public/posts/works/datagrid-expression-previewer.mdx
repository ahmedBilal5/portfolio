Assalam u Alaikum Folks! The motivation for this blog is to show how the **MUI data-grid** React component can be used in a _unique_ use-case: To preview the results of nested expressions. Here is how it looks:
![Basic Expression Preview Example](../images/expandcollapse.gif)

## 1. Type Definitions
 First thing that we need are type definitions for our expression language. These can, of course, be quite comprehensive depending on how complex our language is. A short sample of the types for my language in typescript:
 ```ts
export type Switch = {
  Expression: SwitchExpressionBlock;
  Cases: CaseBlock[];
};
export type SwitchExpressionBlock = {
  AssignedTo: Attribute | undefined;
  SwitchType: 'attribute' | 'expression' | 'function';
  SwitchExpression: Attribute | ExpressionTree | Function | undefined;
};
export type CaseBlock = {
  LeftOperand: Constant;
  RightOperandType: 'attribute' | 'constant' | 'expression' | 'function' | 'null';
  RightOperand: Attribute | Constant | ExpressionTree | Function | undefined;
};

// If else types
export type IfElse = {
  IfBlocks: IfBlock[];
  ElseBlock: ThenBlock;
};

export type IfBlock = {
  ConditionType: 'function' | 'expression';
  Condition: Function | ExpressionTree | undefined;
  Then: ThenBlock;
};

export type ThenBlock = {
  LeftOperand: Attribute | undefined;
  RightOperandType: 'attribute' | 'constant' | 'expression' | 'function' | 'null';
  RightOperand: Attribute | Constant | ExpressionTree | Function | undefined;
  ExprResult?: (string | string[] | number | {})[]; //For Preview value result
};

export type ExpressionTree = Expression[];

export type Expression = {
  Operator: string;
  ExpressionType: 'attribute' | 'expression' | 'constant' | 'function' | 'null';
  ExpressionNode: Attribute | Constant | Function | ExpressionTree | undefined;
  ExprResult?: (string | string[] | number | {})[]; //For Preview value result
};
 ```
 This is just a subset of all the types to get the main idea across. Note that we will be using the data-grid only to **display** the expression and its computed results at each level, not **build** the expression. Building is done separately using an expression builder component which is out of the scope of our topic.

## 4. Building the Expression Trees
This is an interesting part. We will need to build the right data structure to pass to the `data-grid` `rows` prop. As you can imagine from our type definitions in Step 1, we will need to recursively traverse the structures to build the tree data. For that purpose, I decided to write a custom hook to encapsulate all the recursive logic. I will provide part of it here to give you a good idea:
```ts
//For Mapping Preview
export enum PreviewTreeTokenType {
  PARAM = 'param',
  CONSTANT = 'constant',
  ATTRIBUTE = 'attribute',
  NULL = 'null',
  OPENING_BRACKET = '(',
  CLOSING_BRACKET = ')',
  ELLIPSES = '...',
  FUNCTION_ELLIPSES = 'functionEllipses',
  ASSIGN = '=',
  THEN = 'then',
  EXPRESSION_OPERATOR = 'expressionOperator',
  EXPRESSION = 'expression',
  IFCONDITION = 'ifCondition',
  ELSEIFCONDITION = 'elseIfCondition',
  ELSECONDITION = 'elseCondition',
}

export type PreviewFunctionTreeType = {
  treePath: string;
  type:
    | Exclude<
        PreviewTreeTokenType,
        PreviewTreeTokenType.ELLIPSES | PreviewTreeTokenType.FUNCTION_ELLIPSES
      >
    | FunctionType;
  source?:
    | Attribute
    | Constant
    | Function
    | IfBlock
    | ThenBlock
    | ExpressionTree
    | { [key: string]: any }
    | string;
  value?: string | string[] | number | {};
};

const useConstructFunctionTree = () => {
  /**
     Function to extract and return all source values from mappingJson
    */
  const extractMappingsSource = (mappingJson: Mapping[]) => {
    const mappingJsonSourceOnly = mappingJson.map(
      (mapping: Mapping) => mapping.Source
    );
    return mappingJsonSourceOnly;
  };

  /**
     Function to construct tree data for mapping preview
    */
  const constructTree = (mappingJson: Mapping[]) => {
    const functionTree: PreviewFunctionTreeType[] = [];
    const mappingJsonSourcesOnly = extractMappingsSource(mappingJson).filter(
      mappingSource => mappingSource !== undefined
    );
    mappingJsonSourcesOnly.forEach((mappingSource, index) => {
      if (mappingSource) {
        //case of function
        if ('InputColumns' in mappingSource) {
          functionTree.push(
            ...getTreeForFunction(mappingSource, mappingSource?.id)
          );
        }
        //case of attribute or constant
        else {
          functionTree.push(
            ...getTreeForAttribute(
              mappingSource,
              `${index}-${mappingSource?.Path}`
            )
          );
        }
      }
    });
    return functionTree;
  };

  const getTreeForFunction = (mappingFunction: Function, treePath: string) => {
    const functionTree: PreviewFunctionTreeType[] = [];
    functionTree.push({
      treePath: `${treePath}`,
      source: mappingFunction,
      type: mappingFunction?.Type,
      value: mappingFunction?.ExprResult?.[0],
    });
    if (mappingFunction?.Conditions) {
      if ('IfBlocks' in mappingFunction?.Conditions)
        functionTree.push(
          ...getTreeForIfConditionalFunction(
            mappingFunction?.Conditions,
            treePath
          )
        );
      else
        functionTree.push(
          ...getTreeForSwitchConditionalFunction(
            mappingFunction?.Conditions,
            treePath
          )
        );
    } else {
      functionTree.push(
        ...getTreeForFunctionParams(mappingFunction?.Params, treePath)
      );
      functionTree.push(
        ...getTreeForFunctionInputs(mappingFunction?.InputColumns, treePath)
      );
    }
    functionTree.push({
      treePath: `${treePath}/)`, //adding path for closing bracket
      type: PreviewTreeTokenType.CLOSING_BRACKET,
    });
    return functionTree;
  };

  const getTreeForAttribute = (
    attribute: Attribute | Constant,
    treePath: string
  ) => {
    const functionTree: PreviewFunctionTreeType[] = [];
    if ('IsUserDefined' in attribute) {
      functionTree.push({
        treePath: `${treePath}`,
        source: attribute,
        type: PreviewTreeTokenType.CONSTANT,
        value: attribute?.ExprResult?.[0],
      });
    } else {
      functionTree.push({
        treePath: `${treePath}`,
        source: attribute,
        type: PreviewTreeTokenType.ATTRIBUTE,
        value: attribute?.ExprResult?.[0],
      });
    }
    return functionTree;
  };
  
```

We have defined the following types of tokens that we will be catering:

```typescript

CONSTANT  =  'constant',
ATTRIBUTE  =  'attribute',

CLOSING_BRACKET  =  ')',
```

> Remember that each row, when collapsed, will show the whole expression (in a compressed form) that is formed down its tree. But when we expand a row, it only shows the token at its current tree path, be that a function, attribute, bracket or any other.

![Expression expanded and Collapsed](../images/expandcollapse.gif)

Therefore, aside from creating the tree path hierarchy for the data-grid, we also need a way to calculate what we display on each row, when collapsed and when expanded.  For that, we will add a second capability to our hook `useConstructFunctionTree`, that is, computing the token chain for each row. Here is the main function for that:
```typescript
 /**
     Function to construct array of tokens for mapping preview
    */
  const constructTokenArray = (
    treeNode: Pick<PreviewFunctionTreeType, 'source' | 'type'>,
    isExpanded?: boolean
  ): (PreviewToken | PreviewFunctionToken)[] => {
    const tokenArray: (PreviewToken | PreviewFunctionToken)[] = [];
    let node = treeNode?.source;
    const type = treeNode?.type;
    switch (type) {
      case FunctionType.FUNCTION:
      case FunctionType.OPERATOR: {
        tokenArray.push(
          {
            name:
              (node as Function)?.Name === 'If'
                ? 'If Conditional'
                : (node as Function)?.Name,
            type: (node as Function)?.Type,
            depthLevel: 0,
          },
          {
            type: PreviewTreeTokenType.OPENING_BRACKET,
          }
        );
        if (!isExpanded) {
          tokenArray.push(
            ...constructTokenArrayForFunctionInputs(node as Function)
          );
          tokenArray.push({
            type: PreviewTreeTokenType.CLOSING_BRACKET,
          });
        }
        break;
      }
      case PreviewTreeTokenType.IFCONDITION:
      case PreviewTreeTokenType.ELSEIFCONDITION:
      case PreviewTreeTokenType.ELSECONDITION: {
        tokenArray.push(
          ...constructTokenArrayForIfConditions(
            isExpanded,
            type,
            node as IfBlock
          )
        );
        break;
      }
      case PreviewTreeTokenType.THEN: {
        tokenArray.push(
          ...getTokensForThenBlock(false, isExpanded, node as ThenBlock)
        );
        break;
      }
      case PreviewTreeTokenType.EXPRESSION: {
        tokenArray.push(
          ...constructTokenArrayForExpression(
            false,
            isExpanded,
            node as ExpressionTree
          )
        );
        break;
      }
      case PreviewTreeTokenType.EXPRESSION_OPERATOR: {
        tokenArray.push({
          name: node as string,
          type: PreviewTreeTokenType.EXPRESSION_OPERATOR,
        });
        break;
      }
      default:
        tokenArray.push(
          ...constructTokenArrayForLeafEntity({
            source: node as any,
            type: type as PreviewTreeTokenType,
          })
        );
        break;
    }

    return tokenArray;
  };
  return { constructTree, constructTokenArray };
};
```
You can get an idea just from the function above how we are using the same recursive approach we used to construct the tree here as well. Exactly what we do in each subfunction is not hard to guess once you get the big picture.

There is a small complication, however. When we use the ```renderCell``` prop to render the cell, we have to completely define ourselves what is to be rendered in an empty cell. This means that for ```tree Data```, the expansion and collapse button that appears by default in grouping column cells, will also not be present when we use ```renderCell```. You can read more about this in [this github issue](https://github.com/mui/mui-x/issues/13400#issuecomment-2154388702). Luckily, there is a workaround for this issue, which is to render a custom grouping column cell rendering our own expand and collapse button. So that is exactly what we will do:
```typescript
const FunctionTreeCustomGroupingColumnCell = ({
  params,
  treeApiRef,
}: {
  params: GridRenderCellParams;
  treeApiRef: React.MutableRefObject<GridApiPro>;
}) => {
  const DEFAULT_INDENT = 4;
  const { id, field, rowNode, value } = params;
  const filteredDescendantCountLookup = useGridSelector(
    treeApiRef,
    gridFilteredDescendantCountLookupSelector
  );
  const filteredDescendantCount =
    filteredDescendantCountLookup[rowNode.id] ?? 0;
  const indentLeft =
    value?.type === PreviewTreeTokenType.CLOSING_BRACKET
      ? DEFAULT_INDENT * rowNode?.depth - DEFAULT_INDENT
      : DEFAULT_INDENT * rowNode?.depth;

  const handleClick = (event: any) => {
    treeApiRef.current.setRowChildrenExpansion(id, !rowNode.childrenExpanded);
    treeApiRef.current.setCellFocus(id, field);
    event.stopPropagation();
  };

  return (
    <BoxComponent
      ml={indentLeft}
      sx={{ display: 'flex', flexDirection: 'row' }}
    >
      <BoxComponent sx={{ width: '28px', mr: '0.5rem' }}>
        {filteredDescendantCount > 1 && (
          <IconButton onClick={handleClick} size="small" disableRipple sx={{}}>
            <ExpandMoreIcon
              sx={{
                transform:
                  'childrenExpanded' in rowNode && rowNode.childrenExpanded
                    ? 'none'
                    : 'rotate(270deg)',
                width: '16px',
                height: '16px',
              }}
            />
          </IconButton>
        )}
      </BoxComponent>
      <FunctionTreeTokenChain tokenChain={value?.tokenChain} />
    </BoxComponent>
  );
};
```

```typescript
const FunctionTree = ({ open }: { open: boolean }) => {
  const apiRef = useGridApiRef();
  const mappingJson = useSelector(
    (state: RootState) => state?.DataFlow?.mapping?.mappingJson
  );
  const sampleData = useSelector(
    (state: RootState) => state?.DataFlow?.sampleData?.data
  );
  const [loading, setLoading] = useState(false);
  const [page, setPage] = useState<number>(0);
  const [functionTree, setFunctionTree] = useState<PreviewFunctionTreeType[]>(
    []
  );
  
  const { constructTree: constructFunctionTree, constructTokenArray } =
    useConstructFunctionTree();

  // Transform Preview uses transform_preview.wasm (Go WASM) only — it does not
  // touch SQLite, so do not gate on the SharedWorker SQLite readiness signal.
  const transformWasm = useTransformWasmStatus();
  const engineReady = transformWasm.ready;
  const engineError = transformWasm.error;


  const fetchMappingResultsOnSampleData = async (
    sampleData: any[],
    recordNumber: number,
    mappingJson: Mapping[]
  ) => {
    try {
      let res = null;
      let fetchedData: Mapping[] = [];
      setLoading(true);
      console.log('mappingJson: ', mappingJson);
      res = await loadWasmAndProcess({
        mappingJson: mappingJson,
        sampleData: [sampleData[recordNumber]],
      });
      fetchedData = res?.data;
      const tree: PreviewFunctionTreeType[] =
        constructFunctionTree(fetchedData);
      setFunctionTree(tree);
    } catch (error: any) {
      console.error('Error fetching values for transforms:', error);
      notification.error({ message: `Error fetching values for transforms` });
      const tree: PreviewFunctionTreeType[] =
        constructFunctionTree(mappingJson);
      setFunctionTree(tree);
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    if (sampleData && mappingJson && open) {
      fetchMappingResultsOnSampleData(sampleData, page, mappingJson);
    }
  }, [sampleData, mappingJson, page, open, engineReady]);

  const customPreviewValueColumnHeader = () => {
    const sampleDataLength = sampleData?.length;
    const upperBound = (sampleDataLength ?? 50)-1;
    const lowerBound = 0

    const setPageNext = () => {
      if (page <= upperBound) {
        setPage(page + 1);
      }
    };
    const setPagePrevious = () => {
      if (page > lowerBound) {
        setPage(page - 1);
      }
    };
    
    return (
      <BoxComponent
        sx={{
          display: 'flex',
          flexDirection: 'row',
          justifyContent: 'space-between',
          flex: 1,
        }}
      >
        <span style={{ fontSize: '0.75rem', fontWeight: '700' }}>VALUES</span>
        <BoxComponent>
          <IconButton
            disableRipple
            disabled={page <= lowerBound}
            onClick={setPagePrevious}
          >
            <img
              src={page <= lowerBound || mappingJson?.length === 0? ForwardIconDisabled : ForwardIcon}
              style={{ transform: 'rotate(180deg)' }}
              alt='page previous'
            />
          </IconButton>
          <IconButton disableRipple disabled={page >= upperBound} onClick={setPageNext}>
            <img src={page >= upperBound  || mappingJson?.length === 0? ForwardIconDisabled : ForwardIcon} alt='page next'/>
          </IconButton>
        </BoxComponent>
      </BoxComponent>
    );
  };

  const columns: GridColDef[] = [
    {
      field: 'value',
      flex: 1,
      disableColumnMenu: true,
      filterable: false,
      sortable: false,
      renderHeader: customPreviewValueColumnHeader,
      renderCell: (params: GridRenderCellParams) => {
          return <SpecialValue value={params?.row?.value}/>
      },
    }
  ];

  const groupingColDef: DataGridProProps['groupingColDef'] = {
    headerName: 'Transformation'.toUpperCase(),
    flex: 2,
    valueGetter: (params: GridValueGetterParams) => {
      const tokenChain = constructTokenArray(
        { source: params?.row?.source, type: params?.row?.type },
        params?.rowNode?.childrenExpanded
      );
      return { tokenChain, type: params?.row?.type };
    },
    renderCell: (params: GridRenderCellParams) => (
      <FunctionTreeCustomGroupingColumnCell
        params={params}
        treeApiRef={apiRef}
      />
    ),
  };

  return (
    <EngineStateGate ready={engineReady} loading={!engineReady && !engineError} error={engineError}>
      <TreeComponent
        apiRef={apiRef}
        rows={functionTree ?? []}
        columns={columns}
        groupingColDef={groupingColDef}
        getRowId={(row: PreviewFunctionTreeType) => row?.treePath}
        getTreeDataPath={(row: PreviewFunctionTreeType) =>
          row?.treePath?.split('/')
        }
        loading={loading}
        headerHeight={45}
        autoHeight={false}
        isRowSelectable={() => false}
        treeViewStyles={FunctionTreeStyles.previewTreeStyles}
        components={{
          NoRowsOverlay: () => (
              <NoRowsOverlay
              message="Create a Mapping to see preview"
              styles={{
                  position: 'absolute',
              }}
              />
          ),
      }}
      />
    </EngineStateGate>
  );
```
